# Datetime On Multi-timezone Servers

For portable timestamp, use:

from datetime import datetime

ts = datetime.now().timestamp()

ts will be same for every computer regardless of their timezone setting.

For portable MariaDb data with type DATETIME, use Python naive datetime:

from datetime import datetime

dt = datetime.utcnow().isoformat(sep=' ')

dt will same for every computer regardless of their timezone setting. It is in UTC timezone.

from datetime import datetime, timezone

datetime.now().timestamp() will result correct UTC timestamp regardless of timezone.

datetime.now().timestamp() != datetime.utcnow().timestamp()

datetime.now().timestamp() == datetime.utcnow().replace(tzinfo=timezone.utc).timestamp()

datetime.now() is naive local date and time

datetime.utcnow() is naive utc date and time

NOTE

naive means without timezone information.

# Socket Server Example

#!/usr/bin/env python3

import socket

server = socket.socket()
server.bind(('', 4321))
server.listen(1)
server.settimeout(10)

done = False

while not done:
    try:
        conn = None
        print('Waiting client...')
        conn, addr = server.accept()
        print('Connected by', addr)
        conn.settimeout(5)

        while True:
            data = conn.recv(1024)
            if not data:
                print('Client disconnected!')
                break
            print(data)
            if data == b'q\n':
                done = True
                conn.shutdown(socket.SHUT_RDWR)
                break
    except KeyboardInterrupt:
        print('Keyboard interrupt.')
        done = True
    except socket.timeout:
        print('Time out.')

    if conn is not None: conn.close()

server.close()
