# TAR and encrypt files in a directory

tar -czf - ./dirname | openssl enc -e -aes256 -md sha256 -out dirname.tar.gz.enc
openssl enc -d -aes256 -md sha256 -in dirname.tar.gz.enc | tar -xz

openssl enc -e -aes256 -md sha256 -in file.txt -out file.txt.enc
openssl enc -d -aes256 -md sha256 -in file.txt.enc -out original-file.txt

# Decrypt and extract credential files

# Multiboot Flash Disk

Flash disk is /dev/sdb

sudo mount -t ext4 /dev/sdb1 /mnt
sudo mkdir /mnt/boot
sudo grub-install --target=i386-pc --recheck --boot-directory=/mnt/boot /dev/sdb

# Various OS Tasks

Change directories and files mode from read-only to read-write.

find ./ -type d -exec chmod 0755 '{}' \;
find ./ -type f -exec chmod 0644 '{}' \;

mount -o loop,offset=xxxx /path/disk.img /mnt/disk.img.partition

# Boot ISO File From Hardisk

Notes

  • ISO files are stored in /os-images/ directory on /dev/sda4 partition.
  • After files have been changed, run sudo update-grub command.

# Knoppix : /etc/grub.d/15_knoppix

#!/bin/sh
exec tail -n +3 $0
# This file provides an easy way to add custom menu entries.  Simply type the
# menu entries you want to add after this comment.  Be careful not to change
# the 'exec tail' line above.

menuentry "Knoppix" {
    insmod loopback
    insmod part_msdos
    insmod ext2
    set root='hd0,msdos4'
    loopback loop (hd0,msdos4)/os-images/KNOPPIX_V8.2-2018-05-10-EN.iso
    bootoptions="bootfrom=/dev/sda4/os-images/KNOPPIX_V8.2-2018-05-10-EN.iso lang=en apm=power-off nomce libata.force=noncq hpsa.hpsa_allow_any=1 loglevel=1"
    if cpuid -l; then # Check whether CPU is 64-bit
        linux (loop)/boot/isolinux/linux64 $bootoptions
    else
        linux (loop)/boot/isolinux/linux $bootoptions
    fi
    initrd (loop)/boot/isolinux/minirt.gz
}

# Fedora 28 Workstation : /etc/grub.d/16_fedora

#!/bin/sh
exec tail -n +3 $0
# This file provides an easy way to add custom menu entries.  Simply type the
# menu entries you want to add after this comment.  Be careful not to change
# the 'exec tail' line above.

menuentry "Fedora 28 Workstation" {
    insmod loopback
    insmod part_msdos
    insmod ext2
    set root='hd0,msdos4'
    loopback loop (hd0,msdos4)/os-images/Fedora-Workstation-Live-x86_64-28-1.1.iso
    probe --label --set=cd_label (loop)
    bootoptions="iso-scan/filename=/os-images/Fedora-Workstation-Live-x86_64-28-1.1.iso root=live:CDLABEL=$cd_label rd.live.image quiet"
    linux (loop)/isolinux/vmlinuz $bootoptions
    initrd (loop)/isolinux/initrd.img
}
