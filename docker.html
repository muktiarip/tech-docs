# Docker

# Build Base Container Image For Python Application

Note

Make sure runc is in your path:
export PATH=${PATH}:/usr/lib/cri-o-runc/sbin

container=$(buildah from alpine:latest)
buildah run $container -- apk add --update --no-cache python3 py3-paho-mqtt \
  py3-flask py3-cryptography py3-gunicorn
buildah run $container -- pip3 install --no-cache-dir pymysql python_jwt
buildah config --cmd python3 $container
buildah commit $container mypython3
buildah rm $container

# Build Application Image

container=$(buildah from mypython3:3.7.5)
buildah copy $container /application/src/ /opt/app/
buildah config --port 8000 $container
buildah config --workingdir /opt/app $container
buildah config --cmd "gunicorn --bind 0.0.0.0:8000 flask_app:app" $container
buildah commit $container myapp
buildah rm $container

# Create Pod For Grouping Containers

podman pod create --name=podname -p 1880-9000:1880-9000

podman run --name mariadb -e MYSQL_ROOT_PASSWORD=secret \
    --pod podname \
    -v /data/mariadb:/var/lib/mysql \
    -v /config/my.cnf:/etc/mysql/conf.d/mydocker.cnf \
    -d mariadb:latest

podman run --name mqtt --pod podname -d eclipse-mosquitto

podman run --name myapp --pod podname -d myapp_image

After that, just use podman pod start podname and podman pod stop podname commands.

# Delete Left-over winsz Files of Podman Stop Command

find ${CONTAINERS_DIR}/vfs-containers/ -type p -name winsz -exec rm {} \;

# Systemd Unit File For Starting/Stopping Container Pod

[Unit]
Description=Podman in Systemd

[Service]
Restart=on-failure
ExecStartPre=/usr/bin/rm -f /%t/%n-pid /%t/%n-cid
ExecStart=/usr/bin/podman run --conmon-pidfile  **/%t/%n-pid**  --cidfile /%t/%n-cid -d alpine:latest top
ExecStop=/usr/bin/sh -c "/usr/bin/podman rm -f `cat /%t/%n-cid`"
KillMode=none
Type=forking
PIDFile=**/%t/%n-pid**

[Install]
WantedBy=multi-user.target

# Run Docker with user privillege

docker run -u $(id -u):$(id -g) args...

docker run --name tf2 -it \
  -u $(id -u):$(id -g) \
  -v /data/ml_works/content:/content \
  -v /data/ml_works/notebooks:/tf/notebooks \
  -p 8888:8888 \
  tensorflow/tensorflow:latest-py3-jupyter

# Run /bin/bash as root
docker exec -it -u 0:0 tf2 bash

docker start -i -a tf2

# MariaDb on Docker

Run server:

podman run --name mariadb --rm -e MYSQL_ROOT_PASSWORD=mysecret \
  -v /data/mariadb:/var/lib/mysql \
  -v /config/mydocker.cnf:/etc/mysql/conf.d/mydocker.cnf \
  -p 3306:3306 -d mariadb:latest

# Docker Host Networking

docker run \
-d \
--name plex \
--network=host \
-e TZ="<timezone>" \
-e PLEX_CLAIM="<claimToken>" \
-v <path/to/plex/database>:/config \
-v <path/to/transcode/temp>:/transcode \
-v <path/to/media>:/data \
plexinc/pms-docker

# Docker Macvlan Networking

docker run \
-d \
--name plex \
--network=physical \
--ip=<IPAddress> \
-e TZ="<timezone>" \
-e PLEX_CLAIM="<claimToken>" \
-h <HOSTNAME> \
-v <path/to/plex/database>:/config \
-v <path/to/transcode/temp>:/transcode \
-v <path/to/media>:/data \
plexinc/pms-docker

# Build Container Image Of dnsmasq Application

container=$(buildah from alpine:latest)

buildah config --cmd "dnsmasq -d" $container
buildah commit $container dnsmasq
buildah rm $container
